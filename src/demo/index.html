          let nt = new Nativetable('nativetable', {
            sources: data,
            pagination: {

            }
          })

Filtering

Sample code for filtering data 

          let nt = new Nativetable('nativetable', {
            sources: data,
            pagination: {
              maxLength: 5
            },
            filters: {
              $and: {
                name: ['julie', 'sarah'],
                age: (age) => {
                  return age >= 29 && age <= 39
                }
              },
              $or: {
                man: [false],
                brothers: [1]
              }
            }
          })

Sorting

Sample code for sortable table 

          let nt = new Nativetable('nativetable', {
            sources: data,
            sorting: true
          })
